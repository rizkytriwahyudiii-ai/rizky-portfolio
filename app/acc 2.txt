function TypingEffect({ theme }: { theme: 'dark' | 'light' }) {
  const [index, setIndex] = useState(0);
  const [subIndex, setSubIndex] = useState(0);
  const [reverse, setReverse] = useState(false);

  useEffect(() => {
    if (subIndex === words[index].length + 1 && !reverse) {
      const timeout = setTimeout(() => setReverse(true), 2000);
      return () => clearTimeout(timeout);
    }
    if (subIndex === 0 && reverse) {
      setReverse(false);
      setIndex((prev) => (prev + 1) % words.length);
      return;
    }
    const timeout = setTimeout(() => {
      setSubIndex((prev) => prev + (reverse ? -1 : 1));
    }, reverse ? 30 : 70);
    return () => clearTimeout(timeout);
  }, [subIndex, index, reverse]);

  return (
    <span className={theme === 'dark' ? 'text-white' : 'text-slate-900'}>
      {words[index].substring(0, subIndex)}
      <span className={`animate-pulse ${theme === 'dark' ? 'text-amber-500' : 'text-blue-600'}`}>_</span>
    </span>
  );
}

      <section id="contact" className="py-32 px-6">
        <div className={`max-w-4xl mx-auto rounded-[3rem] overflow-hidden border p-1 md:p-2 transition-all duration-700
          ${theme === 'dark' ? 'bg-slate-900/50 border-slate-800 shadow-3xl shadow-black' : 'bg-slate-100 border-slate-200 shadow-xl shadow-slate-200'}`}>
          <div className={`rounded-[2.5rem] grid lg:grid-cols-5 transition-all duration-700
            ${theme === 'dark' ? 'bg-[#030712]' : 'bg-white'}`}>
            
            <div className="lg:col-span-2 p-10 md:p-14 space-y-8 flex flex-col justify-center">
              <h2 className="text-4xl font-black tracking-tighter italic">Let's <br />Connect.</h2>

                <div className="space-y-2">
                  <label className="text-[10px] font-bold uppercase tracking-widest opacity-50">Nama</label>
